Introduction

What is Canvas?

<canvas></canvas>

The canvas element is part of HTML5 and allows for dynamic, scriptable rendering of 2D shapes and bitmap images.

Drawing Context

Canvas provides two drawing contexts:

2D and WebGL (3D).

The 2D context is suitable for simple 2D graphics, while WebGL provides powerful tools for creating 3D scenes and effects.

How to use Canvas?



	function drawRectangle(container: HTMLElement) {

	// Get the drawing context for the canvas
		const canvas = document.createElement("canvas");
		canvas.style.width = '400px';
		canvas.style.height = '200px';
		canvas.style.border = '2px solid white';
		const ctx = canvas.getContext("2d");

	// Set parameters for the rectangle
		const x = 50; // x-coordinate of the top-left corner
		const y = 50; // y-coordinate of the top-left corner
		const width = 150; // width of the rectangle
		const height = 100; // height of the rectangle

	// Draw the rectangle
		ctx.fillStyle = "blue"; // Set the fill color of the
		ctx.fillRect(x, y, width, height); // Use the fillRect()

		container?.append(canvas);
	}

	drawRectangle(document.body);
					

	function drawCircle(container: HTMLElement) {

		const canvas = document.createElement("canvas");
		canvas.setAttribute('width', '250px');
		canvas.setAttribute('height', '200px');
		canvas.style.border = '2px solid white';
		const ctx = canvas.getContext("2d");

		const centerX = canvas.width / 2;
		const centerY = canvas.height / 2;
		const radius = 50;

		ctx.beginPath();
		ctx.arc(centerX, centerY, radius, 0, 2 * Math.PI);
		ctx.fillStyle = "red";
		ctx.fill();
		ctx.strokeStyle = "orange";
		ctx.lineWidth = 2;
		ctx.stroke();

		container?.append(canvas);
	}
	drawCircle(document.body);
									

	function drawLines(container: HTMLElement) {

		const canvas = document.createElement("canvas");
		canvas.setAttribute('width', '250px');
		canvas.setAttribute('height', '200px');
		canvas.style.border = '2px solid white';
		const ctx = canvas.getContext("2d");

		ctx.beginPath();
		ctx.moveTo(50, 50);
		ctx.lineTo(200, 50);
		ctx.strokeStyle = "blue";
		ctx.lineWidth = 3;
		ctx.stroke();

		ctx.beginPath();
		ctx.moveTo(50, 100);
		ctx.lineTo(200, 100);
		ctx.strokeStyle = "green";
		ctx.lineWidth = 5;
		ctx.stroke();

		ctx.beginPath();
		ctx.moveTo(50, 150);
		ctx.lineTo(200, 150);
		ctx.strokeStyle = "orange";
		ctx.lineWidth = 7;
		ctx.stroke();

		container?.append(canvas);
	}
	drawLines(document.body);

Animation on Canvas


	const canvas = document.createElement("canvas");
	canvas.setAttribute('width', '250px');
	canvas.setAttribute('height', '200px');
	canvas.style.border = '2px solid white';
	document.body.append(canvas);

	const ctx = canvas.getContext("2d");
	let x = 0;

	function animateCircle() {
		ctx.clearRect(0, 0, canvas.width, canvas.height);

		const centerX = x;
		const centerY = canvas.height / 2;
		const radius = 50;

		ctx.beginPath();
		ctx.arc(centerX, centerY, radius, 0, 2 * Math.PI);
		ctx.fillStyle = "blue";
		ctx.fill();

		x++

		if (x > canvas.width) {
			x = 0;
		}

		requestAnimationFrame(animateCircle);
	}
	animateCircle();
